}

// backend-go/internal/llm/prompt_sanitizer.go
package llm

import "strings"

// BuildUserContentBlock 使用 sentinel 包裹用户内容，防止 Prompt 注入
// 用户内容会被放在 ###USER_CONTENT_START### 和 ###USER_CONTENT_END### 之间
// LLM Prompt 会被指示忽略该区域之外的内容
func BuildUserContentBlock(nodes []NodeContent) string {
    var sb strings.Builder
    sb.WriteString("###USER_CONTENT_START###\n")

    for i, node := range nodes {
        sb.WriteString(fmt.Sprintf("[节点 %d]\n%s\n---\n", i+1, node.Content))
    }

    sb.WriteString("###USER_CONTENT_END###\n")
    return sb.String()
}

type NodeContent struct {
    Content string
}

// SanitizeContent 基础内容清洗（移除潜在的恶意指令前缀）
func SanitizeContent(content string) string {
    // 移除以某些指令开头的行
    lines := strings.Split(content, "\n")
    var filtered []string
    for _, line := range lines {
        trimmed := strings.TrimSpace(line)
        // 忽略明显的问题指令
        if strings.HasPrefix(strings.ToLower(trimmed), "ignore previous") ||
           strings.HasPrefix(strings.ToLower(trimmed), "forget all") ||
           strings.HasPrefix(strings.ToLower(trimmed), "# system") {
            continue
        }
        filtered = append(filtered, line)
    }
    return strings.Join(filtered, "\n")
}
```

// backend-go/internal/api/http/middleware.go

// AITriggerRateLimiter AI触发API限流：每用户≤10次/分钟
func AITriggerRateLimiter(redisClient *redis.Client) func(http.Handler) http.Handler {
    return func(next http.Handler) http.Handler {
        return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
            user := GetUserFromContext(r.Context())
            if user == nil {
                next.ServeHTTP(w, r)
                return
            }

            key := fmt.Sprintf("ratelimit:ai_trigger:%s", user.ID)
            count, err := redisClient.Incr(r.Context(), key).Result()
            if err != nil {
                // Redis失败时放行，避免影响正常请求
                next.ServeHTTP(w, r)
                return
            }

            // 设置过期时间
            if count == 1 {
                redisClient.Expire(r.Context(), key, time.Minute)
            }

            if count > 10 {
                writeError(w, http.StatusTooManyRequests, "rate limit exceeded: max 10 AI triggers per minute")
                return
            }

            next.ServeHTTP(w, r)
        })
    }
}
```

// 生成幂等性key：基于客户端UUID + 用户ID + 时间戳
function generateIdempotencyKey(): string {
  const stored = sessionStorage.getItem('ai_trigger_idem_key')
  if (stored) return stored

  const key = `${Date.now()}-${Math.random().toString(36).substr(2, 9)}`
  sessionStorage.setItem('ai_trigger_idem_key', key)
  return key
}
