```go
// backend-go/internal/api/http/handlers_usage_alert.go
package httpapi

import (
    "encoding/json"
    "net/http"
    "time"

    "github.com/chenqilscy/ripple/backend-go/internal/service"
    "github.com/go-chi/chi/v5"
)

type UsageAlertHandlers struct {
    Svc  *service.UsageAlertService
    Orgs *service.OrgService
}

type usageAlertResp struct {
    OrgID            string     `json:"org_id"`
    ThresholdPercent int        `json:"threshold_percent"`
    Enabled          bool       `json:"enabled"`
    LastTriggeredAt  *time.Time `json:"last_triggered_at,omitempty"`
}

func toUsageAlertResp(a *service.UsageAlert) usageAlertResp {
    return usageAlertResp{
        OrgID:            a.OrgID,
        ThresholdPercent: a.ThresholdPercent,
        Enabled:          a.Enabled,
        LastTriggeredAt:  a.LastTriggeredAt,
    }
}

// GetUsageAlert GET /api/v1/organizations/{id}/usage-alert
func (h *UsageAlertHandlers) GetUsageAlert(w http.ResponseWriter, r *http.Request) {
    u, ok := CurrentUser(r.Context())
    if !ok {
        writeError(w, http.StatusUnauthorized, "unauthorized")
        return
    }
    orgID := chi.URLParam(r, "id")

    if h.Orgs != nil {
        isMember, err := h.Orgs.IsMember(r.Context(), u.ID, orgID)
        if err != nil || !isMember {
            writeError(w, http.StatusForbidden, "access denied")
            return
        }
    }

    alert, err := h.Svc.GetAlert(r.Context(), orgID)
    if err != nil {
        writeError(w, http.StatusInternalServerError, "failed to get usage alert")
        return
    }
    if alert == nil {
        // 返回默认值
        writeJSON(w, http.StatusOK, usageAlertResp{
            OrgID:            orgID,
            ThresholdPercent: 80,
            Enabled:          false,
        })
        return
    }
    writeJSON(w, http.StatusOK, toUsageAlertResp(alert))
}

type usageAlertReq struct {
    ThresholdPercent int  `json:"threshold_percent"`
    Enabled          bool `json:"enabled"`
}

// UpdateUsageAlert PUT /api/v1/organizations/{id}/usage-alert
func (h *UsageAlertHandlers) UpdateUsageAlert(w http.ResponseWriter, r *http.Request) {
    u, ok := CurrentUser(r.Context())
    if !ok {
        writeError(w, http.StatusUnauthorized, "unauthorized")
        return
    }
    orgID := chi.URLParam(r, "id")

    // 权限校验：必须是组织管理员
    if h.Orgs != nil {
        role, err := h.Orgs.GetMemberRole(r.Context(), orgID, u.ID)
        if err != nil || role != "owner" {
            writeError(w, http.StatusForbidden, "only org owner can manage usage alert")
            return
        }
    }

    r.Body = http.MaxBytesReader(w, r.Body, 16*1024)
    var in usageAlertReq
    if err := json.NewDecoder(r.Body).Decode(&in); err != nil {
        writeError(w, http.StatusBadRequest, "invalid json")
        return
    }

    if in.ThresholdPercent < 0 || in.ThresholdPercent > 100 {
        writeError(w, http.StatusBadRequest, "threshold_percent must be between 0 and 100")
        return
    }

    alert, err := h.Svc.UpdateAlert(r.Context(), orgID, in.ThresholdPercent, in.Enabled)
    if err != nil {
        writeError(w, http.StatusInternalServerError, "failed to update usage alert")
        return
    }
    writeJSON(w, http.StatusOK, toUsageAlertResp(alert))
}
```

```tsx
interface PromptTemplateOption {
  id: string
  name: string
  description: string
  scope: 'private' | 'org'
}

// 从模板字符串中提取变量名
function extractVariables(template: string): string[] {
  const matches = template.match(/\{\{(\w+)\}\}/g) || []
  return [...new Set(matches.map(m => m.slice(2, -2)))]
}

export default function AiTriggerButton({ lakeId, nodeId, inputNodeIds, onDone, onFail }: Props) {
  const [showDropdown, setShowDropdown] = useState(false)
  const [templates, setTemplates] = useState<PromptTemplateOption[]>([])
  const [selectedTemplate, setSelectedTemplate] = useState<PromptTemplateOption | null>(null)
  const [loadingTemplates, setLoadingTemplates] = useState(false)
  const dropdownRef = useRef<HTMLDivElement>(null)

  // 加载模板列表
  useEffect(() => {
    if (showDropdown && templates.length === 0) {
      setLoadingTemplates(true)
      api.listPromptTemplates({ limit: 100 })
        .then(res => setTemplates(res.items.map(t => ({
          id: t.id,
          name: t.name,
          description: t.description || '',
          scope: t.scope,
        }))))
        .catch(() => setTemplates([]))
        .finally(() => setLoadingTemplates(false))
    }
  }, [showDropdown])

  // 点击外部关闭下拉
  useEffect(() => {
    function handleClickOutside(e: MouseEvent) {
      if (dropdownRef.current && !dropdownRef.current.contains(e.target as Node)) {
        setShowDropdown(false)
      }
    }
    document.addEventListener('mousedown', handleClickOutside)
    return () => document.removeEventListener('mousedown', handleClickOutside)
  }, [])

  // 分组模板
  const privateTemplates = templates.filter(t => t.scope === 'private')
  const orgTemplates = templates.filter(t => t.scope === 'org')

  const handleTemplateSelect = (template: PromptTemplateOption) => {
    setSelectedTemplate(template)
    setShowDropdown(false)
  }

  const handleTrigger = useCallback(async () => {
    if (triggering || (job && (job.status === 'pending' || job.status === 'processing'))) return
    // ... 原有逻辑，使用 selectedTemplate?.id 作为 prompt_template_id
  }, [/* dependencies */])

  return (
    <div style={{ position: 'relative', display: 'inline-flex', flexDirection: 'column', gap: 'var(--space-sm)', alignItems: 'flex-start' }}>
      <Button
        variant="primary"
        size="sm"
        onClick={() => setShowDropdown(!showDropdown)}
        disabled={disabled}
        title="选择Prompt模板并触发AI处理"
      >
        <span style={{ fontSize: 'var(--font-lg)' }}>✦</span>
        {selectedTemplate ? selectedTemplate.name : 'AI触发'}
      </Button>

      {/* 模板下拉 */}
      {showDropdown && (
        <div
          ref={dropdownRef}
          style={{
            position: 'absolute',
            top: '100%',
            left: 0,
            zIndex: 1000,
            background: 'var(--bg-surface)',
            border: '1px solid var(--border)',
            borderRadius: 'var(--radius-lg)',
            boxShadow: 'var(--shadow-lg)',
            minWidth: 280,
            maxHeight: 320,
            overflow: 'auto',
            marginTop: 'var(--space-xs)',
          }}
        >
          {loadingTemplates ? (
            <div style={{ padding: 'var(--space-md)', color: 'var(--text-tertiary)' }}>加载中…</div>
          ) : (
            <>
              {privateTemplates.length > 0 && (
                <div>
                  <div style={{ padding: 'var(--space-sm) var(--space-md)', fontSize: 'var(--font-sm)', color: 'var(--text-tertiary)', fontWeight: 600 }}>
                    📋 私有模板
                  </div>
                  {privateTemplates.map(t => (
                    <TemplateOption key={t.id} template={t} onSelect={handleTemplateSelect} />
                  ))}
                </div>
              )}
              {orgTemplates.length > 0 && (
                <div>
                  <div style={{ padding: 'var(--space-sm) var(--space-md)', fontSize: 'var(--font-sm)', color: 'var(--text-tertiary)', fontWeight: 600 }}>
                    📂 组织共享
                  </div>
                  {orgTemplates.map(t => (
                    <TemplateOption key={t.id} template={t} onSelect={handleTemplateSelect} />
                  ))}
                </div>
              )}
              {templates.length === 0 && (
                <div style={{ padding: 'var(--space-md)', color: 'var(--text-tertiary)' }}>
                  暂无可用模板，请在设置中创建
                </div>
              )}
            </>
          )}
        </div>
      )}

      {/* 已选模板信息 */}
      {selectedTemplate && (
        <div style={{ fontSize: 'var(--font-sm)', color: 'var(--text-secondary)' }}>
          <span style={{ color: 'var(--accent)' }}>✓</span>
          已选择: {selectedTemplate.name}
        </div>
      )}
    </div>
  )
}

        cursor: 'pointer',
        borderBottom: '1px solid var(--border)',
      }}
      onClick={() => onSelect(template)}
    >
      <div style={{ fontWeight: 600, color: 'var(--text-primary)' }}>{template.name}</div>
      <div style={{ fontSize: 'var(--font-sm)', color: 'var(--text-tertiary)' }}>
        {template.description || '无描述'}
      </div>
    </div>
  )
}
```
