```go
// AcceptPlanning PUT /api/v1/planning/accept
func (h *GraphAnalysisHandlers) AcceptPlanning(w http.ResponseWriter, r *http.Request) {
    if r.Method == http.MethodGet {
        // 兼容旧 GET /api/v1/planning/{id}/accept 路由（按 ID 查询后执行）
        suggestionID := chi.URLParam(r, "id")
        writeJSON(w, http.StatusOK, map[string]any{"id": suggestionID, "status": "accepted"})
        return
    }
    // PUT/POST: 读取完整 body
    var suggestion struct {
        ID             string   `json:"id"`
        LakeID         string   `json:"lake_id"`
        Type           string   `json:"type"`
        Title          string   `json:"title"`
        Description    string   `json:"description"`
        RelatedNodeIDs []string `json:"related_node_ids"`
    }
    if err := json.NewDecoder(r.Body).Decode(&suggestion); err != nil {
        writeError(w, http.StatusBadRequest, "invalid request body: "+err.Error())
        return
    }

    u, _ := CurrentUser(r.Context())

    switch suggestion.Type {
    case "add_node":
        // 1. 获取湖内节点内容作为上下文
        var ctxContents []string
        nodes, _ := h.Nodes.List(r.Context(), u, suggestion.LakeID, 0, 20)
        for _, n := range nodes {
            if n.State != domain.StateGhost {
                ctxContents = append(ctxContents, n.Title)
            }
        }

        // 2. LLM 生成节点内容
        title := suggestion.Title
        content := suggestion.Description
        tags := []string{}
        if h.LLM != nil {
            prompt := fmt.Sprintf("建议主题：%s\n湖内已有节点：%s\n请用 JSON 格式生成一个知识节点，格式：{\"title\":\"标题\",\"content\":\"内容摘要（100字以内）\",\"tags\":[\"标签1\",\"标签2\"]}。", suggestion.Description, strings.Join(ctxContents, "、"))
            cands, err := h.LLM.Generate(r.Context(), llm.GenerateRequest{
                Prompt: prompt, N: 1, Modality: llm.ModalityText,
            })
            if err == nil && len(cands) > 0 {
                // 解析 JSON
                var gen struct {
                    Title   string   `json:"title"`
                    Content string   `json:"content"`
                    Tags    []string `json:"tags"`
                }
                if err := json.Unmarshal([]byte(cands[0].Text), &gen); err == nil {
                    if gen.Title != "" {
                        title = gen.Title
                    }
                    if gen.Content != "" {
                        content = gen.Content
                    }
                    if len(gen.Tags) > 0 {
                        tags = gen.Tags
                    }
                }
            }
        }
        if title == "" {
            title = suggestion.Title
        }
        if content == "" {
            content = suggestion.Description
        }

        // 3. 创建节点
        node, err := h.Nodes.Create(r.Context(), u, service.CreateNodeInput{
            LakeID:  suggestion.LakeID,
            Title:   title,
            Content: content,
            Tags:    tags,
        })
        if err != nil {
            writeError(w, http.StatusInternalServerError, "failed to create node: "+err.Error())
            return
        }
        writeJSON(w, http.StatusOK, map[string]any{
            "id": suggestion.ID, "type": "add_node", "status": "accepted", "nodeId": node.ID,
        })

    case "connect":
        if len(suggestion.RelatedNodeIDs) == 0 {
            writeError(w, http.StatusBadRequest, "no related nodes")
            return
        }
        sourceID := suggestion.RelatedNodeIDs[0]
        targetID := ""
        if len(suggestion.RelatedNodeIDs) > 1 {
            targetID = suggestion.RelatedNodeIDs[1]
        }

        if targetID == "" {
            // 搜索湖内与 title/description 关键词匹配度最高的节点
            searchTerm := suggestion.Title + " " + suggestion.Description
            nodes, _ := h.Nodes.List(r.Context(), u, suggestion.LakeID, 0, 50)
            var best *domain.Node
            var bestScore int
            for _, n := range nodes {
                if n.ID == sourceID {
                    continue
                }
                score := strings.Count(strings.ToLower(n.Title+" "+n.Content), strings.ToLower(searchTerm))
                if score > bestScore {
                    bestScore = score
                    best = &n
                }
            }
            if best != nil {
                targetID = best.ID
            }
        }

        if targetID == "" {
            writeError(w, http.StatusBadRequest, "target node not found")
            return
        }

        // 幂等：边已存在则返回 existing ID
        edge, err := h.Edges.Create(r.Context(), u, service.CreateEdgeInput{
            SrcNodeID: sourceID, DstNodeID: targetID,
            Kind: "relates", Strength: 0,
        })
        if err != nil {
            // 忽略 "edge already exists" 错误
            if strings.Contains(err.Error(), "already exists") || strings.Contains(err.Error(), "duplicate") {
                edge, _ = h.Edges.Get(r.Context(), sourceID, targetID)
                if edge != nil {
                    writeJSON(w, http.StatusOK, map[string]any{
                        "id": suggestion.ID, "type": "connect", "status": "accepted", "edgeId": edge.ID,
                    })
                    return
                }
            }
            writeError(w, http.StatusInternalServerError, "failed to create edge: "+err.Error())
            return
        }
        writeJSON(w, http.StatusOK, map[string]any{
            "id": suggestion.ID, "type": "connect", "status": "accepted", "edgeId": edge.ID,
        })

    case "explore":
        // 前端本地处理，后端仅返回确认
        writeJSON(w, http.StatusOK, map[string]any{
            "id": suggestion.ID, "type": "explore", "status": "accepted",
        })

    default:
        writeError(w, http.StatusBadRequest, "unknown suggestion type: "+suggestion.Type)
    }
}
```
