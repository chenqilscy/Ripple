```
用户点击"AI触发" → 弹出模板选择下拉 → 选择模板 → 确认触发
```

### 2.2 前端交互设计

- 在 `AiTriggerButton` 中集成模板选择下拉

### 2.3 数据来源

```typescript
// 加载模板列表
async function loadTemplates(): Promise<PromptTemplateOption[]> {
  const res = await api.listPromptTemplates({ limit: 100 })
  return res.items.map(t => ({
    id: t.id,
    name: t.name,
    description: t.description || '',
    scope: t.scope,
    variables: extractVariables(t.template),  // 从模板中提取 {{变量}}
  }))
}

// 从模板字符串中提取变量名
function extractVariables(template: string): string[] {
  const matches = template.match(/\{\{(\w+)\}\}/g) || []
  return [...new Set(matches.map(m => m.slice(2, -2)))]
}
```

## 三、D.2 用量告警

### 3.1 核心流程

```
用户设置阈值（80%配额）→ 后台检查用量 → 超阈值时发送通知 → 前端显示警告
```

### 3.2 API 设计

**新增端点：**

```go
// GET /api/v1/organizations/{id}/usage-alert
// 获取当前告警设置
type UsageAlertResp struct {
    OrgID            string     `json:"org_id"`
    ThresholdPercent int        `json:"threshold_percent"` // 0-100
    Enabled          bool       `json:"enabled"`
    LastTriggeredAt *time.Time `json:"last_triggered_at,omitempty"`
}

// PUT /api/v1/organizations/{id}/usage-alert
// 更新告警设置
type UsageAlertReq struct {
    ThresholdPercent int  `json:"threshold_percent"` // 0-100，如 80
    Enabled          bool `json:"enabled"`
}
```

**错误码：**
| 端点 | 错误码 | 场景 |
|------|--------|------|
| GET /usage-alert | 404 | 组织无告警设置（返回默认值） |
| PUT /usage-alert | 400 | threshold_percent 超出 0-100 范围 |
| PUT /usage-alert | 403 | 非组织管理员 |

### 3.3 数据模型

```sql
-- org_usage_alerts 表（新）
CREATE TABLE org_usage_alerts (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    org_id UUID NOT NULL REFERENCES organizations(id),
    threshold_percent INT NOT NULL DEFAULT 80 CHECK (threshold_percent >= 0 AND threshold_percent <= 100),
    enabled BOOLEAN NOT NULL DEFAULT true,
    last_triggered_at TIMESTAMP WITH TIME ZONE,
    created_at TIMESTAMP WITH TIME ZONE NOT NULL DEFAULT NOW(),
    updated_at TIMESTAMP WITH TIME ZONE NOT NULL DEFAULT NOW(),
    UNIQUE(org_id)
);

-- 告警触发历史（可选，用于展示告警历史）
CREATE TABLE usage_alert_history (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    org_id UUID NOT NULL REFERENCES organizations(id),
    triggered_at TIMESTAMP WITH TIME ZONE NOT NULL DEFAULT NOW(),
    used_percent INT NOT NULL,
    threshold_percent INT NOT NULL
);
```

### 3.4 后端检查逻辑

```go
// 在 LLMUsageService 中添加告警检查
func (s *LLMUsageService) CheckAndSendAlert(ctx context.Context, orgID string) error {
    alert, err := s.GetAlertSetting(ctx, orgID)
    if err != nil {
        return err
    }
    if !alert.Enabled {
        return nil
    }

    // 获取当月用量
    usage, err := s.GetOrgUsage(ctx, orgID, 30) // 近30天
    if err != nil {
        return err
    }

    // 获取配额
    quota, err := s.GetOrgQuota(ctx, orgID)
    if err != nil {
        return err
    }

    // 计算使用百分比
    usedPercent := 0
    if quota.AICallsPerMonth > 0 {
        usedPercent = int(float64(usage.TotalCalls) / float64(quota.AICallsPerMonth) * 100)
    }

    // 检查是否触发告警（仅每月第一天检查上月是否超限）
    if usedPercent >= alert.ThresholdPercent {
        // 发送通知
        notification := &domain.Notification{
            Type:       domain.NotificationTypeUsageAlert,
            OrgID:      orgID,
            Title:      "AI用量告警",
            Content:    fmt.Sprintf("您的AI调用量已达到配额的 %d%%", usedPercent),
            Metadata:   map[string]any{"used_percent": usedPercent, "threshold": alert.ThresholdPercent},
        }
        s.NotificationService.Create(ctx, notification)

        // 更新最后触发时间
        s.UpdateLastTriggered(ctx, orgID, time.Now())

        // 记录历史
        s.RecordAlertHistory(ctx, orgID, usedPercent, alert.ThresholdPercent)
    }

    return nil
}
```

**告警触发时机：**
- 用户每次调用 AI 触发 API 时检查
- 或每日定时任务检查（更推荐，避免频繁查询）

**告警设置对话框：**

```tsx
function UsageAlertConfigModal({ orgId, currentSettings, onSave }) {
  const [threshold, setThreshold] = useState(currentSettings.threshold_percent)
  const [enabled, setEnabled] = useState(currentSettings.enabled)

  return (
    <Modal title="AI用量告警设置">
      <div>
        <label>
          <input
            type="checkbox"
            checked={enabled}
            onChange={e => setEnabled(e.target.checked)}
          />
          启用用量告警
        </label>
      </div>

      {enabled && (
        <div>
          <label>告警阈值：</label>
          <select value={threshold} onChange={e => setThreshold(Number(e.target.value))}>
            <option value={50}>50%</option>
            <option value={80}>80%</option>
            <option value={90}>90%</option>
            <option value={100}>100%</option>
          </select>
          <span> 当用量达到此百分比时发送通知</span>
        </div>
      )}

      <Button onClick={() => onSave({ threshold_percent: threshold, enabled })}>
        保存设置
      </Button>
    </Modal>
  )
}
```
